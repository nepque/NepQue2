    <script>
      document.addEventListener("DOMContentLoaded", function () {
        // Filter tabs functionality
        const filterTabs = document.querySelectorAll(".filter-tab");
        filterTabs.forEach((tab) => {
          tab.addEventListener("click", function () {
            // Remove active class from all tabs
            filterTabs.forEach((t) => t.classList.remove("active"));
            filterTabs.forEach((t) =>
              t.classList.add("bg-gray-100", "text-gray-700"),
            );
            filterTabs.forEach((t) => t.classList.remove("bg-primary", "text-white"));

            // Add active class to clicked tab
            this.classList.add("active");
            this.classList.remove("bg-gray-100", "text-gray-700");
            this.classList.add("bg-primary", "text-white");

            // Filter coupons based on category
            const category = this.getAttribute("data-category");
            const couponCards = document.querySelectorAll(".coupon-card");

            let visibleCount = 0;

            couponCards.forEach((card) => {
              if (
                category === "all" ||
                card.getAttribute("data-category") === category
              ) {
                card.style.display = "block";
                visibleCount++;
              } else {
                card.style.display = "none";
              }
            });

            // Show/hide empty state
            const emptyState = document.getElementById("emptyState");
            const couponsGrid = document.getElementById("couponsGrid");

            if (visibleCount === 0) {
              emptyState.classList.remove("hidden");
              couponsGrid.classList.add("hidden");
            } else {
              emptyState.classList.add("hidden");
              couponsGrid.classList.remove("hidden");
            }
          });
        });

        // Coupon modal functionality
        const showCouponBtns = document.querySelectorAll(".show-coupon-btn");
        const couponModal = document.getElementById("couponModal");
        const closeModal = document.getElementById("closeModal");
        const couponCode = document.getElementById("couponCode");
        const copyCode = document.getElementById("copyCode");
        const copiedMessage = document.querySelector(".copied-message");

        showCouponBtns.forEach((btn) => {
          btn.addEventListener("click", function () {
            const code = this.getAttribute("data-code");
            couponCode.textContent = code;
            couponModal.style.display = "flex";
          });
        });

        closeModal.addEventListener("click", function () {
          couponModal.style.display = "none";
        });

        copyCode.addEventListener("click", function () {
          const textToCopy = couponCode.textContent;
          navigator.clipboard.writeText(textToCopy).then(function () {
            copiedMessage.classList.add("show");
            setTimeout(function () {
              copiedMessage.classList.remove("show");
            }, 2000);
          });
        });

        // Close modal when clicking outside
        window.addEventListener("click", function (event) {
          if (event.target === couponModal) {
            couponModal.style.display = "none";
          }
        });

        // Filter modal functionality
        const filterButton = document.getElementById("filterButton");
        const filterModal = document.getElementById("filterModal");
        const closeFilterModal = document.getElementById("closeFilterModal");
        const applyFilters = document.getElementById("applyFilters");
        const clearFilters = document.getElementById("clearFilters");

        filterButton.addEventListener("click", function () {
          filterModal.style.display = "flex";
        });

        closeFilterModal.addEventListener("click", function () {
          filterModal.style.display = "none";
        });

        applyFilters.addEventListener("click", function () {
          filterModal.style.display = "none";
          // Simulate loading state
          showLoadingState();

          setTimeout(() => {
            hideLoadingState();
          }, 1000);
        });

        clearFilters.addEventListener("click", function () {
          const checkboxes = filterModal.querySelectorAll('input[type="checkbox"]');
          checkboxes.forEach((checkbox) => {
            checkbox.checked = true;
          });
        });

        // Close filter modal when clicking outside
        window.addEventListener("click", function (event) {
          if (event.target === filterModal) {
            filterModal.style.display = "none";
          }
        });

        // Reset filters button
        const resetFilters = document.getElementById("resetFilters");
        resetFilters.addEventListener("click", function () {
          const allTab = document.querySelector('.filter-tab[data-category="all"]');
          allTab.click();
        });

        // Load more button
        const loadMoreBtn = document.getElementById("loadMoreBtn");
        loadMoreBtn.addEventListener("click", function () {
          showLoadingState();

          setTimeout(() => {
            hideLoadingState();
            loadMoreBtn.textContent = "No More Coupons";
            loadMoreBtn.disabled = true;
            loadMoreBtn.classList.add("opacity-50", "cursor-not-allowed");
          }, 1500);
        });

        // Loading state functions
        function showLoadingState() {
          document.getElementById("loadingState").classList.remove("hidden");
          document.getElementById("couponsGrid").classList.add("hidden");
          document.getElementById("emptyState").classList.add("hidden");
        }

        function hideLoadingState() {
          document.getElementById("loadingState").classList.add("hidden");
          document.getElementById("couponsGrid").classList.remove("hidden");
        }
      });
    </script>
  </body>
</html>
