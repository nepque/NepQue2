        const closeModal = document.getElementById("closeModal");
        const couponCode = document.getElementById("couponCode");
        const copyCode = document.getElementById("copyCode");
        const goToStore = document.getElementById("goToStore");
        const copiedMessage = document.querySelector(".copied-message");
        showCouponBtns.forEach((btn) => {
          btn.addEventListener("click", function () {
            const code = this.getAttribute("data-code");
            couponCode.textContent = code;
            couponModal.style.display = "flex";
            // Set the store URL (in a real app, this would be dynamic)
            const storeName = this.closest(".deal-card")
              .querySelector("h3")
              .textContent.trim()
              .toLowerCase();
            goToStore.href = `https://www.${storeName}.com`;
          });
        });
        closeModal.addEventListener("click", function () {
          couponModal.style.display = "none";
        });
        copyCode.addEventListener("click", function () {
          const textToCopy = couponCode.textContent;
          navigator.clipboard.writeText(textToCopy).then(function () {
            copiedMessage.classList.add("show");
            setTimeout(function () {
              copiedMessage.classList.remove("show");
            }, 2000);
          });
        });
        // Close modal when clicking outside
        window.addEventListener("click", function (event) {
          if (event.target === couponModal) {
            couponModal.style.display = "none";
          }
        });
      });
