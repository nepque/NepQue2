    <script>
      document.addEventListener("DOMContentLoaded", function () {
        // Filter dropdowns
        const filterButtons = document.querySelectorAll(
          "#categoryFilter, #discountFilter, #storeFilter, #expiryFilter, #sortFilter",
        );
        const filterDropdowns = document.querySelectorAll(
          "#categoryDropdown, #discountDropdown, #storeDropdown, #expiryDropdown, #sortDropdown",
        );

        filterButtons.forEach((button, index) => {
          button.addEventListener("click", function () {
            filterDropdowns[index].classList.toggle("show");

            // Close other dropdowns
            filterDropdowns.forEach((dropdown, dropdownIndex) => {
              if (dropdownIndex !== index && dropdown.classList.contains("show")) {
                dropdown.classList.remove("show");
              }
            });
          });
        });

        // Close dropdowns when clicking outside
        window.addEventListener("click", function (event) {
          filterButtons.forEach((button, index) => {
            if (
              !button.contains(event.target) &&
              !filterDropdowns[index].contains(event.target)
            ) {
              filterDropdowns[index].classList.remove("show");
            }
          });
        });

        // Coupon modal functionality
        const showCouponBtns = document.querySelectorAll(".show-coupon-btn");
        const couponModal = document.getElementById("couponModal");
        const closeModal = document.getElementById("closeModal");
        const couponCode = document.getElementById("couponCode");
        const copyCode = document.getElementById("copyCode");
        const goToStore = document.getElementById("goToStore");
        const copiedMessage = document.querySelector(".copied-message");

        showCouponBtns.forEach((btn) => {
          btn.addEventListener("click", function () {
            const code = this.getAttribute("data-code");
            couponCode.textContent = code;
            couponModal.style.display = "flex";

            // Set the store URL based on the closest store section
            const storeSection = this.closest(".coupon-card")
              .closest("div")
              .closest("div").previousElementSibling;
            if (storeSection) {
              const storeName = storeSection
                .querySelector("h2")
                .textContent.trim()
                .toLowerCase();
              goToStore.href = `https://www.${storeName}.com`;
            }
          });
        });

        closeModal.addEventListener("click", function () {
          couponModal.style.display = "none";
        });

        copyCode.addEventListener("click", function () {
          const textToCopy = couponCode.textContent;
          navigator.clipboard.writeText(textToCopy).then(function () {
            copiedMessage.classList.add("show");
            setTimeout(function () {
              copiedMessage.classList.remove("show");
            }, 2000);
          });
        });

        // Close modal when clicking outside
        window.addEventListener("click", function (event) {
          if (event.target === couponModal) {
            couponModal.style.display = "none";
          }
        });
      });
    </script>
