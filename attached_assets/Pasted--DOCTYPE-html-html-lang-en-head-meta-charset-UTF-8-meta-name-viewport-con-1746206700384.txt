      document.addEventListener("DOMContentLoaded", function () {
        // Filter Modal
        const filterButton = document.getElementById("filterButton");
        const filterModal = document.getElementById("filterModal");
        const closeFilterModal = document.getElementById("closeFilterModal");
        const resetFilters = document.getElementById("resetFilters");
        const applyFilters = document.getElementById("applyFilters");

        filterButton.addEventListener("click", function () {
          filterModal.style.display = "flex";
        });

        closeFilterModal.addEventListener("click", function () {
          filterModal.style.display = "none";
        });

        resetFilters.addEventListener("click", function () {
          const checkboxes = filterModal.querySelectorAll('input[type="checkbox"]');
          checkboxes.forEach((checkbox) => {
            checkbox.checked = false;
          });

          const numberInputs = filterModal.querySelectorAll('input[type="number"]');
          numberInputs.forEach((input) => {
            input.value = "";
          });
        });

        applyFilters.addEventListener("click", function () {
          filterModal.style.display = "none";
          // In a real app, this would filter the coupons based on selections
        });

        // Close modal when clicking outside
        window.addEventListener("click", function (event) {
          if (event.target === filterModal) {
            filterModal.style.display = "none";
          }
        });

        // Load More Button
        const loadMoreBtn = document.getElementById("loadMoreBtn");
        loadMoreBtn.addEventListener("click", function () {
          // In a real app, this would load more coupons
          this.textContent = "Loading...";
          setTimeout(() => {
            this.textContent = "Load More (110 remaining)";
          }, 1000);
        });
      });

      document.addEventListener("DOMContentLoaded", function () {
        // Coupon Modal
        const showCouponBtns = document.querySelectorAll(".show-coupon-btn");
        const couponModal = document.getElementById("couponModal");
        const closeModal = document.getElementById("closeModal");
        const couponCode = document.getElementById("couponCode");
        const copyCode = document.getElementById("copyCode");
        const goToStore = document.getElementById("goToStore");
        const copiedMessage = document.querySelector(".copied-message");

        showCouponBtns.forEach((btn) => {
          btn.addEventListener("click", function () {
            const code = this.getAttribute("data-code");
            couponCode.textContent = code;
            couponModal.style.display = "flex";

            // Set the store URL (in a real app, this would be dynamic)
            const storeName = this.closest(".coupon-card")
              .querySelector("h3")
              .textContent.trim()
              .toLowerCase();
            goToStore.href = `https://www.${storeName}.com`;
          });
        });

        closeModal.addEventListener("click", function () {
          couponModal.style.display = "none";
        });

        copyCode.addEventListener("click", function () {
          const textToCopy = couponCode.textContent;
          navigator.clipboard.writeText(textToCopy).then(function () {
            copiedMessage.classList.add("show");
            setTimeout(function () {
              copiedMessage.classList.remove("show");
            }, 2000);
          });
        });

        // Close modal when clicking outside
        window.addEventListener("click", function (event) {
          if (event.target === couponModal) {
            couponModal.style.display = "none";
          }
        });
      });
